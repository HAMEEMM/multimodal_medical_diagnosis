import pandas as pd
import numpy as np

from sklearn.pipeline import Pipeline
from sklearn.metrics import classification_report, confusion_matrix, accuracy_score
from sklearn.ensemble import RandomForestClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.svm import LinearSVC
from sklearn.naive_bayes import MultinomialNB
import nltk
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer
from nltk.tokenize import word_tokenize
from textblob import TextBlob
import warnings
warnings.filterwarnings('ignore')

# Download necessary NLTK data
nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')

print("Libraries imported successfully")

# 2.2 LOAD AND PREPROCESS DATA
# ------------------------------------------------
print("\nLoading dataset...")
# Use relative path to locate the dataset file
data_path = "data/overview-of-recordings.csv"
data = pd.read_csv(data_path)
print(f"Dataset shape: {data.shape}")
print(f"Number of unique prompts: {data['prompt'].nunique()}")
print(f"Prompt classes: {data['prompt'].unique()}")

# Check for class imbalance
print("\nClass distribution:")
class_distribution = data['prompt'].value_counts()
print(class_distribution)

# Plot class distribution
plt.figure(figsize=(10, 6))
sns.barplot(x=class_distribution.values, y=class_distribution.index)
plt.title("Distribution of Medical Prompt Classes")
plt.xlabel("Number of Samples")
plt.tight_layout()
plt.show()

# Text preprocessing function with lemmatization
def preprocess_text(text):
    if pd.isna(text):
        return ""
    
    # Lowercase
    text = text.lower()
    
    # Remove special characters
    text = ''.join([c for c in text if c.isalnum() or c.isspace()])
    
    # Tokenize
    tokens = word_tokenize(text)
    
    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    tokens = [token for token in tokens if token not in stop_words]
    
    # Lemmatize
    lemmatizer = WordNetLemmatizer()
    tokens = [lemmatizer.lemmatize(token) for token in tokens]
    
    # Join back to string
    return ' '.join(tokens)

# Apply preprocessing
print("\nPreprocessing text data...")
data['processed_text'] = data['phrase'].apply(preprocess_text)

# Display a sample of processed text
print("\nSample of processed text:")
for i, row in data.sample(3).iterrows():
    print(f"Original: {row['phrase']}")
    print(f"Processed: {row['processed_text']}")
    print()

# ------------------------------------------------
# 2.3 FEATURE EXPLORATION
# ------------------------------------------------
# Extract additional text features
print("\nExtracting additional text features...")
data['text_length'] = data['processed_text'].apply(len)
data['word_count'] = data['processed_text'].apply(lambda x: len(x.split()))
data['sentiment'] = data['phrase'].apply(lambda x: TextBlob(str(x)).sentiment.polarity if pd.notna(x) else np.nan)
data['subjectivity'] = data['phrase'].apply(lambda x: TextBlob(str(x)).sentiment.subjectivity if pd.notna(x) else np.nan)

# Plot additional feature distributions
plt.figure(figsize=(16, 12))

plt.subplot(2, 2, 1)
sns.boxplot(x='prompt', y='text_length', data=data)
plt.title("Text Length by Prompt")
plt.xticks(rotation=45)

plt.subplot(2, 2, 2)
sns.boxplot(x='prompt', y='word_count', data=data)
plt.title("Word Count by Prompt")
plt.xticks(rotation=45)

plt.subplot(2, 2, 3)
sns.boxplot(x='prompt', y='sentiment', data=data)
plt.title("Sentiment by Prompt")
plt.xticks(rotation=45)

plt.subplot(2, 2, 4)
sns.boxplot(x='prompt', y='subjectivity', data=data)

# ------------------------------------------------
# 2.4 DATA SPLITTING FOR TRAINING AND TESTING
# ------------------------------------------------
print("\nSplitting data into training and testing sets...")
# Extract the core features and target
X_text = data['processed_text']
X_additional = data[['text_length', 'word_count', 'sentiment', 'subjectivity']]
y = data['prompt']

# Split the data with stratification
X_train_text, X_test_text, y_train, y_test = train_test_split(
    X_text, y, test_size=0.2, random_state=42, stratify=y
)

X_train_add, X_test_add, _, _ = train_test_split(
    X_additional, y, test_size=0.2, random_state=42, stratify=y
)

print(f"Training set size: {len(X_train_text)}")
print(f"Testing set size: {len(X_test_text)}")

# ------------------------------------------------
# 2.5 MODEL BUILDING AND EVALUATION - TF-IDF BASED
# ------------------------------------------------
print("\n--- TF-IDF BASED MODELS ---")

# Define TF-IDF pipeline
tfidf_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(max_features=5000, ngram_range=(1, 2))),
])

# Transform training and testing data
X_train_tfidf = tfidf_pipeline.fit_transform(X_train_text)
X_test_tfidf = tfidf_pipeline.transform(X_test_text)

print(f"TF-IDF features shape: {X_train_tfidf.shape}")

# 2.5.1 Logistic Regression with TF-IDF
print("\nTraining Logistic Regression model...")
lr_model = LogisticRegression(max_iter=1000, C=10, solver='liblinear', multi_class='ovr')
lr_model.fit(X_train_tfidf, y_train)
lr_predictions = lr_model.predict(X_test_tfidf)

print("\nLogistic Regression Results:")
lr_accuracy = accuracy_score(y_test, lr_predictions)
print(f"Accuracy: {lr_accuracy:.4f}")
print(classification_report(y_test, lr_predictions))

# 2.5.2 Random Forest with TF-IDF
print("\nTraining Random Forest model...")
rf_model = RandomForestClassifier(n_estimators=100, random_state=42)
rf_model.fit(X_train_tfidf, y_train)
rf_predictions = rf_model.predict(X_test_tfidf)

print("\nRandom Forest Results:")
rf_accuracy = accuracy_score(y_test, rf_predictions)
print(f"Accuracy: {rf_accuracy:.4f}")
print(classification_report(y_test, rf_predictions))

# 2.5.3 Support Vector Machine with TF-IDF
print("\nTraining SVM model...")
svm_model = LinearSVC(C=1.0, random_state=42, max_iter=10000)
svm_model.fit(X_train_tfidf, y_train)
svm_predictions = svm_model.predict(X_test_tfidf)

print("\nSVM Results:")
svm_accuracy = accuracy_score(y_test, svm_predictions)
print(f"Accuracy: {svm_accuracy:.4f}")
print(classification_report(y_test, svm_predictions))

# 2.5.4 Naive Bayes with TF-IDF
print("\nTraining Naive Bayes model...")
nb_model = MultinomialNB()
nb_model.fit(X_train_tfidf, y_train)
nb_predictions = nb_model.predict(X_test_tfidf)

print("\nNaive Bayes Results:")
nb_accuracy = accuracy_score(y_test, nb_predictions)
print(f"Accuracy: {nb_accuracy:.4f}")
print(classification_report(y_test, nb_predictions))

# ------------------------------------------------
# 2.6 CONFUSION MATRIX ANALYSIS
# ------------------------------------------------
# Identify the best performing model
print("\nIdentifying the best performing model...")
accuracies = {
    'Logistic Regression': lr_accuracy,
    'Random Forest': rf_accuracy,
    'SVM': svm_accuracy,
    'Naive Bayes': nb_accuracy
}

best_model_name = max(accuracies, key=accuracies.get)
best_predictions = None

if best_model_name == 'Logistic Regression':
    best_predictions = lr_predictions
elif best_model_name == 'Random Forest':
    best_predictions = rf_predictions
elif best_model_name == 'SVM':
    best_predictions = svm_predictions
else:
    best_predictions = nb_predictions

print(f"\nBest model: {best_model_name} with accuracy: {accuracies[best_model_name]:.4f}")

# Plot confusion matrix for the best model
plt.figure(figsize=(14, 12))
cm = confusion_matrix(y_test, best_predictions)
sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
            xticklabels=data['prompt'].unique(),
            yticklabels=data['prompt'].unique())
plt.xlabel('Predicted')
plt.ylabel('True')
plt.title(f'Confusion Matrix: {best_model_name}')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# ------------------------------------------------
# 2.7 FEATURE IMPORTANCE ANALYSIS
# ------------------------------------------------
print("\nAnalyzing feature importance...")

# Get the TF-IDF vectorizer
tfidf_vectorizer = tfidf_pipeline.named_steps['tfidf']
feature_names = tfidf_vectorizer.get_feature_names_out()

# For each class, get top features (for Logistic Regression or SVM)
if best_model_name in ['Logistic Regression', 'SVM']:
    model = lr_model if best_model_name == 'Logistic Regression' else svm_model
    
    # For multiclass, we have a coefficient vector per class
    if hasattr(model, 'coef_'):
        for i, class_label in enumerate(model.classes_):
            # Get top 15 features for this class
            if model.coef_.shape[0] > i:  # Ensure we don't exceed array bounds
                top_features_idx = np.argsort(model.coef_[i])[-15:]
                top_features = [feature_names[idx] for idx in top_features_idx]
                print(f"\nTop features for '{class_label}':")
                print(", ".join(top_features))

# For Random Forest, feature importance is different
elif best_model_name == 'Random Forest':
    feature_importances = rf_model.feature_importances_
    indices = np.argsort(feature_importances)[-20:]  # Top 20 features
    
    plt.figure(figsize=(10, 8))
    plt.title('Top 20 Feature Importances')
    plt.barh(range(len(indices)), feature_importances[indices], color='b', align='center')
    plt.yticks(range(len(indices)), [feature_names[i] for i in indices])
    plt.xlabel('Relative Importance')
    plt.tight_layout()
    plt.show()

    ('tfidf', TfidfVectorizer(max_features=5000, ngram_range=(1, 2))),
    ('classifier', LogisticRegression(max_iter=1000, C=10, solver='liblinear', multi_class='ovr'))
])

rf_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(max_features=5000, ngram_range=(1, 2))),
    ('classifier', RandomForestClassifier(n_estimators=100, random_state=42))
])

svm_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(max_features=5000, ngram_range=(1, 2))),
    ('classifier', LinearSVC(C=1.0, random_state=42, max_iter=10000))
])

nb_pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(max_features=5000, ngram_range=(1, 2))),
    ('classifier', MultinomialNB())
])

# Define stratified cross-validation
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

# Perform cross-validation for each model
print("\nLogistic Regression cross-validation:")
lr_cv_scores = cross_val_score(lr_pipeline, X_text, y, cv=cv, scoring='accuracy')
print(f"Mean CV Score: {lr_cv_scores.mean():.4f}, Std: {lr_cv_scores.std():.4f}")
print(f"Individual scores: {lr_cv_scores}")

print("\nRandom Forest cross-validation:")
rf_cv_scores = cross_val_score(rf_pipeline, X_text, y, cv=cv, scoring='accuracy')
print(f"Mean CV Score: {rf_cv_scores.mean():.4f}, Std: {rf_cv_scores.std():.4f}")
print(f"Individual scores: {rf_cv_scores}")

print("\nSVM cross-validation:")
svm_cv_scores = cross_val_score(svm_pipeline, X_text, y, cv=cv, scoring='accuracy')
print(f"Mean CV Score: {svm_cv_scores.mean():.4f}, Std: {svm_cv_scores.std():.4f}")
print(f"Individual scores: {svm_cv_scores}")

print("\nNaive Bayes cross-validation:")
nb_cv_scores = cross_val_score(nb_pipeline, X_text, y, cv=cv, scoring='accuracy')
print(f"Mean CV Score: {nb_cv_scores.mean():.4f}, Std: {nb_cv_scores.std():.4f}")
print(f"Individual scores: {nb_cv_scores}")

# Compare CV results

    'Random Forest': rf_cv_scores.mean(),
    'SVM': svm_cv_scores.mean(),
    'Naive Bayes': nb_cv_scores.mean()
